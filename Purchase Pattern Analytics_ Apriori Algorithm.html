import pandas as pd
from mlxtend.frequent_patterns import apriori, association_rules

# Step 1 - Load data
df = pd.read_csv('transactions_clean.csv')
print("Data loaded:", df.shape)

# Step 2 - Create basket
basket = df.groupby(['BillNo', 'Itemname'])['Quantity'] \
           .sum().unstack(fill_value=0)
basket = basket.map(lambda x: 1 if x > 0 else 0)
print("Basket shape:", basket.shape)

# Step 3 - Apriori
frequent_items = apriori(basket, min_support=0.1, use_colnames=True)
frequent_items = frequent_items.sort_values('support', ascending=False)
print("Frequent itemsets found:", len(frequent_items))
print(frequent_items.head(10))

# Step 4 - Association rules
rules = association_rules(frequent_items, metric="lift", min_threshold=1.0)
rules = rules.sort_values('lift', ascending=False)
print("\nTotal rules found:", len(rules))
print(rules[['antecedents','consequents','support','confidence','lift']].head(10))

# Step 5 - Save
rules.to_csv('association_rules.csv', index=False)
print("\nSaved!")
